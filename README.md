```sql
BEGIN
  FOR launch_id IN (
    SELECT ID FROM QAPORTAL.TEST_LAUNCH WHERE REGRESSION = '0'
  ) LOOP

    -- Delete TEST_STEP_RUNs
    DELETE FROM QAPORTAL.TEST_STEP_RUN
    WHERE TEST_RUN_ID IN (
      SELECT ID FROM QAPORTAL.TEST_RUN WHERE TEST_LAUNCH_ID = launch_id
    );

    -- Delete TEST_RUNs
    DELETE FROM QAPORTAL.TEST_RUN WHERE TEST_LAUNCH_ID = launch_id;

    -- Delete TEST_STEP_DATA
    DELETE FROM QAPORTAL.TEST_STEP_DATA
    WHERE TEST_STEP_ID IN (
      SELECT ID FROM QAPORTAL.TEST_STEP
      WHERE TEST_ID IN (
        SELECT TEST_ID FROM QAPORTAL.TEST_RUN WHERE TEST_LAUNCH_ID = launch_id
      )
    );

    -- Delete TEST_PARAMs
    DELETE FROM QAPORTAL.TEST_PARAM WHERE TEST_LAUNCH_ID = launch_id;

    -- Delete TEST_TAGs
    DELETE FROM QAPORTAL.TEST_TAG WHERE TEST_LAUNCH_ID = launch_id;

    -- Delete PERFORMANCE_METRICS
    DELETE FROM QAPORTAL.PERFORMANCE_METRICS WHERE TEST_LAUNCH_ID = launch_id;

    -- Finally, delete the TEST_LAUNCH record
    DELETE FROM QAPORTAL.TEST_LAUNCH WHERE ID = launch_id;

  END LOOP;

  -- Cleanup orphaned FIX
  DELETE FROM QAPORTAL.FIX
  WHERE ID NOT IN (
    SELECT FIX_ID FROM QAPORTAL.TEST_RUN WHERE FIX_ID IS NOT NULL
    UNION
    SELECT FIX_ID FROM QAPORTAL.TEST_STEP_RUN WHERE FIX_ID IS NOT NULL
  );

  -- Cleanup orphaned LOG
  DELETE FROM QAPORTAL.LOG
  WHERE ID NOT IN (
    SELECT LOG_ID FROM QAPORTAL.TEST_RUN WHERE LOG_ID IS NOT NULL
    UNION
    SELECT LOG_ID FROM QAPORTAL.TEST_STEP_RUN WHERE LOG_ID IS NOT NULL
  );

  -- Cleanup orphaned EXCEPTION
  DELETE FROM QAPORTAL.EXCEPTION
  WHERE ID NOT IN (
    SELECT EXCEPTION_ID FROM QAPORTAL.TEST_RUN WHERE EXCEPTION_ID IS NOT NULL
    UNION
    SELECT EXCEPTION_ID FROM QAPORTAL.TEST_STEP_RUN WHERE EXCEPTION_ID IS NOT NULL
  );

  -- Cleanup orphaned TEST
  DELETE FROM QAPORTAL.TEST
  WHERE ID NOT IN (
    SELECT TEST_ID FROM QAPORTAL.TEST_RUN
    UNION
    SELECT TEST_ID FROM QAPORTAL.TEST_PARAM
    UNION
    SELECT TEST_ID FROM QAPORTAL.TEST_TAG
  );

  -- Cleanup orphaned TEST_CLASS
  DELETE FROM QAPORTAL.TEST_CLASS
  WHERE ID NOT IN (
    SELECT TEST_CLASS_ID FROM QAPORTAL.TEST
  );

  -- Cleanup orphaned TEST_FEATURE
  DELETE FROM QAPORTAL.TEST_FEATURE
  WHERE ID NOT IN (
    SELECT TEST_FEATURE_ID FROM QAPORTAL.TEST
  );

END;
/
```
